            .url("https://example.com")

微服务应用

// gRPC服务示例
package main

import (
    "github.com/jingjie/sdk-go"
    pb "path/to/generated"
)

type server struct {
    client *jingjie.Client
}

func (s *server) ScrapeData(ctx context.Context,
    req *pb.ScrapeRequest) (*pb.DataResponse, error) {

    spider := s.client.NewSpider(jingjie.SpiderConfig{
        Name:      req.Name,
        URL:       req.Url,
        Selectors: req.Selectors,
    })

    results, err := spider.Run(ctx)
    if err != nil {
        return nil, err
    }

    return &pb.DataResponse{Data: results}, nil
}

并发处理

urls := []string{
    "https://site1.com",
    "https://site2.com",
    // ...
}

results := make(chan []byte, len(urls))

for _, url := range urls {
    go func(u string) {
        spider := client.NewSpider(jingjie.SpiderConfig{
            URL: u,
            // 配置...
        })
        data, _ := spider.Run(ctx)
        results <- data
    }(url)
}

// 收集结果
var allResults [][]byte
for result := range results {
    allResults = append(allResults, result)
}
