type server struct {
    client *jingjie.Client
}

func (s *server) ScrapeData(ctx context.Context,
    req *pb.ScrapeRequest) (*pb.DataResponse, error) {

    spider := s.client.NewSpider(jingjie.SpiderConfig{
        Name:      req.Name,
        URL:       req.Url,
        Selectors: req.Selectors,
    })

    results, err := spider.Run(ctx)
    if err != nil {
        return nil, err
    }

    return &pb.DataResponse{Data: results}, nil
}
